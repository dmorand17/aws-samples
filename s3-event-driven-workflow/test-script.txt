SOURCE_BUCKET_NAME=<BUCKET>

##########
  CLEAN
##########

# Copy object
aws s3api put-object --bucket SOURCE_BUCKET_NAME \
    --key file_1MB.dat \
    --body ./tmp/file_1MB.dat

# Tag resource CLEAN
aws s3api put-object-tagging --bucket $SOURCE_BUCKET_NAME \
    --key file_1MB.dat \
    --tagging 'TagSet=[{Key=ScanResult,Value=CLEAN}]'
